```c++
#include <iostream>
using namespace std;

struct Node {
    int n;
    Node *next;
};

Node *head, *last;

void addNode(int n)
{
    if (head == NULL) {
        head = new Node();
        head->n = n;
        last = head;
        return;
    }

    last->next = new Node();
    last = last->next;
    last->n = n;
}

int main()
{
    int v[5] = { 4, 8, 7, 6, 9 };

    addNode(4);
    addNode(8);
    addNode(7);
    addNode(6);
    addNode(9);

    for (int i = 0; i < 5; i++) {
        cout << v[i];
    }
    cout << endl;
    for (Node *p = head; p != NULL; p = p->next) {
        cout << p->n;
    }

    return 0;
}
```

used 배열을 탐색 후 0으로 초기화 시키면 -> 모든 경로를 탐색

초기화 시키지 않으면 -> 모든 노드를 탐색





```c++
#include <iostream>
#include <vector>
#include <queue>
using namespace std;

int map[4][4] = {
	{0,1,1,1},
	{0,0,1,1},
	{0,0,0,0},
	{1,0,1,0}
};
int used[4] = {};
int cnt = 0;

void dfs(int n) {
	if (n == 2) cnt++;

	for (int i = 0; i < 4; i++) {
		if (used[i] == 1) continue;
		if (map[n][i] == 0) continue;
		used[i] = 1;
		dfs(i);
	    used[i] = 0;
	}
}

int main()
{
	dfs(0);

	cout << cnt <<"개"<< endl;

	return 0;
}
```

    used[2] = 1;

    while (!q.empty()) {

        int now = q.front();

        cout << name[now] << " ";

        for (int i = 0; i < 7; i++) {
